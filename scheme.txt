code:
	[ line_or_block ... ] EOF

func_decl:
	func_info EOF

func_def:
	func_info block
	func_info // if deleted

func_info:
	[func_modifier...] var_type <func_name> arg_decl
	[func_modifier...] <class_name> arg_decl // ctor
	[func_modifier...] var_type operator override_op arg_decl

member_decl:
	simple_var_type <member_name> EOF

line_or_block:
	block
	line

block:
	{ [ line_or_block ...] }
	
line:
	class/struct <class_name> { [class_item ...] }
	func_def [ ( expr ... ] // can only be used in top block
	if cond line_or_block [ else line_or_block ]
	do line_or_block while cond
	while cond line_or_block
	for ( [ var_type_decl / expr ] ; [ expr ] ; [ expr ] ) line_or_block
	switch ( expr ) { [switch_case ...] }
	var_type_decl ;
	break ;  // must be inside loop/switch block
	return [expr] ;
	expr ;
	;

switch_case:
	case const_item : [switch_case] [line_or_block ...]
	default : [switch_case] [line_or_block ...]

class_item:
	simple_var_type var_decl [, var_decl ...] ;
	func_def

arg_decl:
	([var_type <arg_name> [= const_item] [, ...] )

var_type_decl:
	var_type var_decl [, var_decl ...]

cond:
	( expr ) // expr returns must return bool

var_type:
	simple_var_type [&]
	void
	string
	<class_name>

simple_var_type:
	bool
	char
	int
	float

var_decl:
	<var_name> [ = expr ]
	<var_name> [ -> expr ]

expr<END>:
	expr_part [op expr_part ...] <END>

expr_part:
	[pre_op ...] item / ( expr ) [post_op]

item:
	const_item
	<var_name>
	<func> args
	<class_name> args

const_item:
	<int>
	<float>
	<char>
	<string>
	const
	<var_type>

args:
	( expr [, expr ...] )

pre_op:
	unary_op
	inc_dec_op

post_op:
	index_op
	inc_dec_op

op:
	base_op
	is
	as
	compare_op
	&&
	||
	!
	( expr )
	. <obj_func> args
	. <obj_member>
	[base_op]=
	=
	inc_dec_op

compare_op:
	==
	!=
	>
	>=
	<
	<=

base_op:
	+
	-
	*
	/
	%
	&
	|
	^
	>>
	<<

unary_op:
	+
	-
	!
	~

inc_dec_op:
	++
	--

index_op:
	[ expr ]
	args // functor

override_op:
	compare_op
	base_op
	unary_op
	base_op=
	[]
	()
	cast
	addref // code only
	release // code only

const:
	true
	false
	this

func_modifier:
	implicit
	delete
